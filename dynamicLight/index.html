public String generateAnswer(String toName) {
    StringBuilder result = new StringBuilder();
    result.append("<html>");
    result.append("<body>";
    result.append("<h1>Hello</h1>");
    result.append("<p>");
    result.append("Hello ");
    result.append(toName);
    result.append("</p>");
    result.append("</body>");
    result.append("</html>");
    return result.toString();
}
